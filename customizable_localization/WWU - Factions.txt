defined_text = {
    name = GetAllianceStatus

    text = {
        localisation_key = FACTION_STATUS_ALLIANCE_HEAD
        
        trigger = {
			has_country_modifier = head_of_alliance
        }
    }
    text = {
        localisation_key = FACTION_STATUS_ALLIANCE_MEMBER
        
        trigger = {
			NOT = { has_country_modifier = head_of_alliance }
        }
    }
}
