sha_of_doubt_disaster = {
	potential = {
		capital_scope = {
			continent = continent_pandaria
		}
	}


	can_start = {
		is_at_war = yes
		NOT = { war_score = -25 }
	}
	
	progress = {
		modifier = {
				factor = 1
				NOT = { war_score = -25 }
		}
		modifier = {
				factor = 1
				NOT = { war_score = -45 }
		}
		modifier = {
				factor = 1
				NOT = { war_score = -65 }
		}
		modifier = {
				factor = 1
				NOT = { war_score = -85 }
		}
	}
	
	can_stop = {
		war_score = 0
	}
	
	can_end = {
		is_at_war = no
	}
	
	modifier = {
	
	}

	on_start = 0
	on_end = 0
	
	on_monthly = {
        random_events = { 
                35 = 0
                1 = sha.1	
        }
	}
}

sha_of_fear_disaster = {
	potential = {
		capital_scope = {
			continent = continent_pandaria
		}
	}


	can_start = {
		NOT = { has_country_modifier = is_evil }
		any_neighbor_country = {
			has_country_modifier = is_evil
			total_development = ROOT
			army_size = ROOT
		}
	}
	
	progress = {
		modifier = {
				factor = 1
				NOT = { has_country_modifier = is_evil }
				any_neighbor_country = {
					has_country_modifier = is_evil
					total_development = ROOT
					army_size = ROOT
				}
		}
		modifier = {
				factor = 2
				any_neighbor_country = {
					has_country_modifier = is_evil
					war_with = ROOT
				}
		}
		modifier = {
				factor = 2
				NOT = { manpower_percentage = 0.75 }
		}
	}
	
	can_stop = {
		any_neighbor_country = {
			has_country_modifier = is_evil
			NOT = { army_size = ROOT }
		}
	}
	
	can_end = {
		any_neighbor_country = {
			has_country_modifier = is_evil
			NOT = { total_development = ROOT }
			NOT = { army_size = ROOT }
		}
	}
	
	modifier = {
	
	}

	on_start = 0
	on_end = 0
	
	on_monthly = {
        random_events = { 
                35 = 0
                1 = sha.4	
        }
	}
}
