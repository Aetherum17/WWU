country_decisions = {
    #--------------------------------------
    # Training
    #--------------------------------------
    # Undergo Magical Training
    undergo_magical_training = {
        potential = {
            NOT = { has_ruler_modifier = spellcaster_capabilities }
            NOT = { has_ruler_modifier = magical_training }
        }
        allow = {
            NOT = { has_ruler_modifier = spellcaster_capabilities }
            adm_power = 100
            
            if = {
                limit = {
                    OR = {
                        ruler_has_personality = mage_personality
                        ruler_has_personality = priest_personality
                        ruler_has_personality = warlock_personality
                        ruler_has_personality = druid_personality
                    }
                }
                treasury = 250
            }
            else = {
                treasury = 1000
            }
        }
        effect = {
            add_adm_power = -100
            
            if = {
                limit = {
                    OR = {
                        ruler_has_personality = mage_personality
                        ruler_has_personality = priest_personality
                        ruler_has_personality = warlock_personality
                        ruler_has_personality = druid_personality
                    }
                }
                add_treasury = -250
            }
            else = {
                add_treasury = -1000
            }
            
            add_ruler_modifier = {
                name = magical_training
                duration = 3650
            }
            
            custom_tooltip = MAGIC_UNDERGO_TRAINING
            
            # Play failure event if the ruler isn't a spellcaster by the end
            hidden_effect = {
                country_event = { id = wwu_magic.12 days = 3650 }
            }
            
            # Add free points based on monarch skill
            # Only do it for the first instance of the decision
            hidden_effect = {
                if = {
                    limit = {
                        is_variable_equal = {
                            which = ruler_magic_training
                            value = 0
                        }
                    }
                    if = {
                        limit = {
                            adm = 5
                        }
                        add_magical_training_point = yes
                    }
                    if = {
                        limit = {
                            dip = 5
                        }
                        add_magical_training_point = yes
                    }
                    if = {
                        limit = {
                            mil = 5
                        }
                        add_magical_training_point = yes
                    }
                }
            }
            
        }		
        ai_will_do = {
            factor = 0
        }
    }
    
    # Improve Aptitude
    improve_aptitude = {
        potential = {
            has_ruler_modifier = spellcaster_capabilities
            NOT = { has_legendary_aptitude = yes }
        }
        allow = {
            has_ruler_modifier = spellcaster_capabilities
            adm_power = 250
        }
        effect = {
            add_adm_power = -250
            
            add_aptitude_point = yes
        }		
        ai_will_do = {
            factor = 0
        }
    }
    
    # Improve Finesse
    improve_finesse = {
        potential = {
            has_ruler_modifier = spellcaster_capabilities
            NOT = { has_legendary_finesse = yes }
        }
        allow = {
            has_ruler_modifier = spellcaster_capabilities
            mil_power = 250
        }
        effect = {
            add_mil_power = -250
            
            add_finesse_point = yes
        }		
        ai_will_do = {
            factor = 0
        }
    }
    
    # Improve Concentration
    improve_concentration = {
        potential = {
            has_ruler_modifier = spellcaster_capabilities
            NOT = { has_legendary_concentration = yes }
        }
        allow = {
            has_ruler_modifier = spellcaster_capabilities
            dip_power = 250
        }
        effect = {
            add_dip_power = -250
            
            add_concentration_point = yes
        }		
        ai_will_do = {
            factor = 0
        }
    }
    
    #--------------------------------------
    # Spells
    #--------------------------------------
    invoke_domain_magic = {
        major = yes
        
        potential = {
            has_ruler_modifier = spellcaster_capabilities
        }
        allow = {
            has_mana_1 = yes
            has_ruler_modifier = spellcaster_capabilities
        }
        effect = {
            country_event = { id = wwu_magic.20 }
        }		
        ai_will_do = {
            factor = 0
        }
    }
}
