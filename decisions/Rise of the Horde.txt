country_decisions = {
    succumb_to_fel = {
        potential = {
            is_year = 570
            
			# Is Orc culture
		    culture_group = culture_group_orc
            
            # Capital is in Draenor
			capital_scope = {
				continent = continent_draenor
			}
            
			NOT = { religion = fel }
		}
		allow = {
			stability = 1
            NOT = { religion = fel }
		}
		effect = {
            add_stability = -3
            
			change_religion = fel
            
			every_owned_province = {
                change_religion = fel
            }
		}
		
		ai_will_do = {
			factor = 1
            
            modifier = {
                factor = 0
                
                NOT = { dominant_religion = fel }
            }
		}
    }

    # Allow the player to open the Dark Portal themselves
	construct_the_dark_portal = {
		potential = {
			NOT = { has_global_flag = dark_portal_built }
            
            is_year = 570
            
            # Is Orc culture
		    culture_group = culture_group_orc
            
            # Capital is in Draenor
			capital_scope = {
				continent = continent_draenor
			}
            
            owns_core_province = 3792
            
            # Is Fel religion
            religion = fel
		}
		allow =  {
            owns_core_province = 3792
            
            # Is Fel religion
            religion = fel

            # Cost
            adm_power = 200
            dip_power = 200
            mil_power = 200
		}
		effect = {
            # Cost
            add_adm_power = -200
            add_dip_power = -200
            add_mil_power = -200

            hidden_effect = {
                clr_global_flag = the_dying_times
                clr_global_flag = rise_of_the_horde
            
                set_global_flag = the_first_war
                set_global_flag = dark_portal_built
            }
            
            3792 = {
                add_canal = gp_dark_portal_draenor
            }
		}
		ai_will_do = {
			factor = 1
		}
	}
    
    # Allow the Orc clans to jump into Azeroth
	establish_foothold_at_azeroth = {
		potential = {
            has_global_flag = dark_portal_built
            
            # Is Orc culture
		    culture_group = culture_group_orc
            
            # Capital is in Draenor
			capital_scope = {
				continent = continent_draenor
			}
            
            # Not moved capital already
			NOT = { has_country_flag = moved_capital }
		}
		allow =  {
            # Is Orc Culture
            culture_group = culture_group_orc
            
            # Capital is in Draenor
			capital_scope = {
				continent = continent_draenor
			}
            
            # Is Fel religion
            religion = fel
            
            # Empty provinces in the Blasted Lands
            region_blasted_lands = {
                type = any
                is_empty = yes
            }
            
            # MIL Cost
            mil_power = 200
		}
		effect = {
			hidden_effect = { set_country_flag = moved_capital }
            
            # MIL Cost
            add_mil_power = -200

            # Bonus for moving to Azeroth
			add_country_modifier = {
				name = motivated_for_assault
				duration = 3650
			}
            
            # Default location if first
            if = {
                limit = {
                    1035 = {
                        is_empty = yes
                    }
                }
                1035 = {
                    cede_province = ROOT
                    add_core = ROOT
                    change_culture = ROOT
                    change_religion = ROOT
                    move_capital_effect = yes
                    add_base_tax = 4
                    add_base_production = 4
                    add_base_manpower = 4
                    add_building = wow_fort_1
                }
            }
            # Otherwise select random province in Blasted Lands
            else = {
                if = {
                    limit = {
                        region_blasted_lands = {
                            type = any
                            is_empty = yes
                        }
                    }
                    random_province = {
                        limit = {
                            region = region_blasted_lands 
                            is_empty = yes
                        }
                        cede_province = ROOT
                        add_core = ROOT
                        change_culture = ROOT
                        change_religion = ROOT
                        move_capital_effect = yes
                        add_base_tax = 4
                        add_base_production = 4
                        add_base_manpower = 4
                        add_building = wow_fort_1
                    }
                }
            }
		}
		ai_will_do = {
			factor = 1
		}
	}
    
    # Used to expand ownership in Blasted Lands
    expand_foothold_in_azeroth = {
		potential = {
		    any_owned_province = {
                OR = {
                    region = region_blasted_lands
                    region = region_swamp_of_sorrows
                }
                any_empty_neighbor_province = {
                    is_empty = yes
                }
            }
            religion = fel
		}
		allow =  {
            adm_power = 100
		}
		effect = {
            add_adm_power = -100
            
            random_owned_province = {
                limit = {
                    OR = {
                        region = region_blasted_lands
                        region = region_swamp_of_sorrows
                    }
                    any_empty_neighbor_province = {
                        is_empty = yes
                    }
                }
                random_empty_neighbor_province = {
                    cede_province = ROOT
                    add_core = ROOT
                    change_culture = ROOT
                    change_religion = ROOT
                }
            }
		}
		ai_will_do = {
			factor = 1
		}
	}
}
