# Inactive
achievement_g_argent_champion = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
        NOT = { is_year = 800 }
        religion_group = light_group
		NOT = { has_country_modifier = achievement_argent_champion }
	}

	trigger = {
        religion = argent_light
        NOT = { exists = P11 }
        area_icecrown_interior = {
            type = all
            owned_by = ROOT
            religion = argent_light
        }
	}
}

# Active
achievement_argent_champion = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
        religion_group = light_group
        OR = {
            AND = {
                NOT = { is_year = 800 }
                religion = argent_light
                NOT = { exists = P11 }
                area_icecrown_interior = {
                    type = all
                    owned_by = ROOT
                    religion = argent_light
                }
            }
            has_country_modifier = achievement_argent_champion
        }
	}

	trigger = {
		always = yes
	}
}
