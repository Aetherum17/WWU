achievement_g_uplifted = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
        NOT = { is_year = 800 }
        NOT = { has_country_modifier = uncivilized_nation }
		NOT = { has_country_modifier = achievement_uplifted }
	}

	trigger = {
        custom_trigger_tooltip = {
            tooltip = UNCIVILIZED_NATION_UPLIFT_TT
            calc_true_if = {
                amount = 8
                
                any_subject_country = {
                    has_country_flag = was_or_is_uncivilized_nation
                }
            }
        }
	}
}

achievement_g_one_with_the_trees = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
        NOT = { is_year = 800 }
		NOT = { has_country_modifier = achievement_one_with_the_trees }
        any_owned_province = {
            OR = {
				has_terrain = forest
				has_terrain = woods
			}
        }
	}

	trigger = {
        custom_trigger_tooltip = {
            tooltip = ONE_WITH_THE_TREES_TT
            all_owned_province = { 
                OR = {
                    has_terrain = forest
                    has_terrain = woods
                }
            }
            NOT = {
                any_country = {
                    NOT = { tag = ROOT }
                    total_development = ROOT
                }
            }
        }
	}
}
