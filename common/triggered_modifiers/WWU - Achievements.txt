# Inactive
achievement_g_uplifted = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
        NOT = { has_country_modifier = uncivilized_nation }
		NOT = { has_country_modifier = achievement_uplifted }
	}

	trigger = {
        custom_trigger_tooltip = {
            tooltip = UNCIVILIZED_NATION_UPLIFT_TT
            calc_true_if = {
                amount = 8
                
                any_subject_country = {
                    has_country_flag = was_or_is_uncivilized_nation
                }
            }
        }
	}
}

# Active
achievement_uplifted = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
        NOT = { has_country_modifier = uncivilized_nation }
        OR = {
            calc_true_if = {
                amount = 8
                
                any_subject_country = {
                    has_country_flag = was_or_is_uncivilized_nation
                }
            }
            has_country_modifier = achievement_uplifted
        }
	}

	trigger = {
		always = yes
	}
}

# Unique Achievements
#-----------------------------------------
# These are achievements for single or small groups of nations.
# Inactive
achievement_g_argent_champion = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
        religion_group = light_group
		NOT = { has_country_modifier = achievement_argent_champion }
	}

	trigger = {
        religion = argent_light
        NOT = { exists = P11 }
        area_icecrown_interior = {
            type = all
            owned_by = ROOT
            religion = argent_light
        }
	}
}
