# Inactive
achievement_g_uplifted = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
        NOT = { has_country_modifier = uncivilized_nation }
		NOT = { has_country_modifier = achievement_uplifted }
	}

	trigger = {
        custom_trigger_tooltip = {
            tooltip = UNCIVILIZED_NATION_UPLIFT_TT
            calc_true_if = {
                amount = 8
                
                any_subject_country = {
                    has_country_flag = was_or_is_uncivilized_nation
                }
            }
        }
	}
}

# Active

	potential = {
        has_global_flag = enabled_achievements
		ai = no

                    has_country_flag = was_or_is_uncivilized_nation
                }
            }

        }
	}

		always = yes
	}
}

achievement_g_disturbing_the_peace = {
	potential = {
        has_global_flag = enabled_achievements
		ai = no
		NOT = { has_country_modifier = achievement_disturbing_the_peace }
	}

	trigger = {
        calc_true_if = {
            amount = 10
            
            all_country = {
                war_with = ROOT
                NOT = { war_score = -50 }
            }
        }
	}
}
