
# Presence of C'Thun
presence_of_cthun = {
	potential = {
        has_global_flag = cthun_active
	}

	trigger = {
        NOT = { religion = cthun }
        capital_scope = {
            OR = {
                superregion = northern_kalimdor_superregion
                superregion = central_kalimdor_superregion
                superregion = southern_kalimdor_superregion
            }
        }
	}
    
    yearly_corruption = 0.05
}
