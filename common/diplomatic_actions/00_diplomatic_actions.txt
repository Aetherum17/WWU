# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

allianceaction = {
    condition = {
        tooltip = DIP_INCOMP_EVIL
        
        potential = {
            end_game_tags_blocked = no
			OR = {
				AND = {
					OR = {
						religion = cthun
						religion = nzoth
						religion = yogg_saron
						religion = yshaarj
						religion = twilight_cult
						religion = nightmare
					}
					FROM = {
						NOT = { religion_group = shadow_group }
						NOT = { is_uncivilized = yes }
					}
				}
				AND = {
					religion_group = disorder_group
					FROM = {
						NOT = { religion_group = disorder_group }
						NOT = { is_uncivilized = yes }
					}
				}
				AND = {
					NOT = { religion_group = shadow_group }
					FROM = {
						OR = {
							religion = cthun
							religion = nzoth
							religion = yogg_saron
							religion = yshaarj
							religion = twilight_cult
							religion = nightmare
						}
					}
				}
				AND = {
					NOT = { religion_group = disorder_group }
					FROM = {
						religion_group = disorder_group
					}
				}
				AND = {
					NOT = { religion_group = death_group }
					FROM = {
						religion_group = death_group
					}
				}
			}
		}
        allow = {
            always = no
        }
	}
    condition = {
        tooltip = DIP_INCOMP_SCOURGE
        
        potential = {
            end_game_tags_blocked = no
			OR = {
				AND = {
					religion = cult_of_the_damned
					FROM = {
						NOT = { religion_group = shadow_group }
                        NOT = { religion_group = death_group }
					}
				}
				AND = {
                    NOT = { religion_group = shadow_group }
                    NOT = { religion_group = death_group }
                    
					FROM = {
						religion = cult_of_the_damned
					}
				}
			}
		}
        allow = {

	}
    condition = {
        tooltip = DIP_INCOMP_UNCIVILIZED
        
        potential = {
            end_game_tags_blocked = no
			OR = {
				AND = {
					is_uncivilized = yes
					FROM = {
						NOT = { is_uncivilized = yes }
					}
				}
				AND = {
					NOT = { is_uncivilized = yes }
					NOT = { is_savage = yes }
					NOT = { is_neighbor_of = FROM }
					NOT = { religion_group = shadow_group }
					NOT = { religion_group = death_group }

					}

        }
	}

				AND = {
					OR = {
						has_country_modifier = head_of_alliance
                        has_country_modifier = member_of_alliance
					}

						OR = {
							has_country_modifier = head_of_horde

annexationaction = {
	condition = {
		tooltip = ANNEX_TOO_LOW
		potential = {
			FROM = {
				is_subject_of = ROOT
			}
		}
		allow = {
			FROM = {
				has_opinion = {
					who = ROOT
					value = 190
				}
			}
		}
	}
    
    # Block early annexation by AI nations
    condition = {
		tooltip = AI_BLOCK_ANNEXATION
		potential = {
            ai = yes
			FROM = {
				is_subject_of = ROOT
			}
		}
		allow = {
			is_year = 600
		}
	}
}
