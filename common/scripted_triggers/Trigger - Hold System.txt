is_valid_hold_race = {
    OR = {
        culture_group = culture_group_kobold
        culture_group = culture_group_trogg
        culture_group = culture_group_furbolg
        culture_group = culture_group_quillboar
        culture_group = culture_group_titan
        culture_group = culture_group_earthen
        culture_group = culture_group_dwarven
    }
}

is_invalid_hold_race = {
    NOT = { culture_group = culture_group_kobold }
    NOT = { culture_group = culture_group_trogg }
    NOT = { culture_group = culture_group_furbolg }
    NOT = { culture_group = culture_group_quillboar }
    NOT = { culture_group = culture_group_titan }
    NOT = { culture_group = culture_group_earthen }
    NOT = { culture_group = culture_group_dwarven }
}

is_valid_hold_province = {
    OR = {
        has_terrain = complex
        has_terrain = hills
        has_terrain = mountain
        has_terrain = impassable_mountains
        has_terrain = underground
        has_province_modifier = opportunistic_hold
    }
}

is_hold_province = {
    has_province_modifier = hold_depth_level_0
}

# Hold Levels
is_hold_level_0 = {
    NOT = {
        check_variable = {
            which = province_hold_level
            value = 1
        }
    }
}

is_hold_level_1 = {
    check_variable = {
        which = province_hold_level
        value = 1
    }
