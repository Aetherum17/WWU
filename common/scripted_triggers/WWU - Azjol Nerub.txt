has_required_cost_for_tunnel_improvement = {
    adm_power = 50
    dip_power = 50
    mil_power = 50
}

is_max_tunnel_depth = {
    check_variable = {
        which = nerubian_tunnel_$NUMBER$_depth
        value = 10
    }
}
