is_uncivilized = {
    custom_trigger_tooltip = {
        tooltip = IS_UNCIVILIZED_TT
        
        has_country_flag = is_uncivilized_nation
    }
}

is_civilized = {
    custom_trigger_tooltip = {
        tooltip = IS_CIVILIZED_TT
        
        OR = { 
            has_country_flag = is_civilized_nation # former unciv 
            NOT = { has_country_flag = is_uncivilized_nation } # default civ
        }
    }
}
