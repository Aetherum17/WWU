# Age Ability Unlock Tiers
is_tier_2_ability = {
    # Trouble in Azeroth or later
    NOT = { current_age = age_rise_of_the_horde }
    NOT = { current_age = age_frozen_throne }
}

is_tier_3_ability = {
    # Wrath of the Lich King or later
    NOT = { current_age = age_rise_of_the_horde }
    NOT = { current_age = age_frozen_throne }
    NOT = { current_age = age_trouble_in_azeroth }
    NOT = { current_age = age_burning_crusade }
}
