#---------------------------------
# Adventurer
#---------------------------------
adventurer_rebels = {
	
	color = { 190 55 55 }
	
	area = nation 					# Stays within the nation
	government = any		# Don't care about gov type				
	defection = none				# Will not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	will_relocate = yes	

	resilient = yes					# Can retreat
	reinforcing = yes				# Can reinforce
	general = yes					# Does use generals
	smart = yes						# Do assault forts, tactically withdraw from battle, etc.
	unit_transfer = no				# Units do not remain after enforcing demands.
	
	artillery = 0.0
	infantry = 0.7
	cavalry = 0.3
	
	morale = 0.75
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 14
		modifier = {
			factor = 0
			NOT = { owner = { has_country_modifier = is_evil } }		
		}
		modifier = {
			factor = 1.25
			is_capital = yes		
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 2.0
			is_capital = yes
		}	
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
	
 	# Province Scope
 	siege_won_trigger = {
 		NOT = { local_autonomy = 50 }
		has_terrain = steppe
 	}
	siege_won_effect = {
		add_local_autonomy = 10	
	}	
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}

	# Localisation for their demands
	demands_description = "adventurer_rebels_demand"
	
	# Country scope
	demands_enforced_effect = {
		add_country_modifier = { name = just_lost_to_rebels duration = 10 hidden = yes }
		if = {
			limit = {
				corruption = 15
			}
			while = {
				limit = {
					NOT = { corruption = 10 }
				}
				random_owned_province = {
					add_base_tax = -1
					add_base_production = -1
					add_base_manpower = -1
				}
				add_corruption = -1
			}
		}
		if = {
			limit = {
				overextension_percentage = 2
			}
			collapse_nation = yes
		}
		if = {
			limit = {
				has_reform = criminal_organization_reform
			}
			exile_ruler_as = ruler_exiled_by_adventurers
			remove_government_reform = criminal_organization_reform
			set_country_flag = ruler_exiled_by_adventurers_flag
			
		}
		if = {
			limit = {
				has_country_modifier = summoned_oldgod
			}
			remove_country_modifier = summoned_oldgod
			add_stability = -1
		}
		if = {
			limit = {
				has_country_modifier = lich_kings_crown
			}
			remove_country_modifier = lich_kings_crown
			add_stability = -1
		}
	}
}

#---------------------------------
# Scourge
#---------------------------------
scourge_rebels = {
	color = { 32 12 164 }

	area = nation 					# Stays within the nation
	government = any
	defection = none                # Will defect to the "best" alternative.
	independence = none				# Does not strive for independence
	defect_delay = 120
	gfx_type = culture_province
	
	resilient = yes					# Disbands on retreat
	reinforcing = yes				# Does not receive reinforcements
	general = yes					# Does not use generals
	smart = yes						# Does not assault forts, tactically withdraw from battle, etc.
	unit_transfer = no				# Units do not remain after enforcing demands.
	will_relocate = yes				# Units will relocate if they have sieged all provinces nearby and has to move over sea.
	
	artillery = 0.0
	infantry = 0.9
	cavalry = 0.1
    
	morale = 0.75

	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no		
 
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 10
        
        modifier = {
            factor = 0
            
            NOT = { has_global_flag = scourge_of_lordaeron }
        }
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			OR = {
                region = region_eastern_plaguelands
                region = region_western_plaguelands
                region = region_tirisfal_glades
            }
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
 		always = yes
 	}
	siege_won_effect = {
        if = {
            limit = {
                NOT = { culture = culture_scourge }
            }
        }
        change_culture = culture_scourge
        change_religion = cult_of_the_damned
        
        # Clean up rebels in Scourge nations
        if = {
            limit = {
                owner = {
                    has_reform = scourge_domain_reform
                }
            }
            kill_units = { who = REB }
            add_yearly_manpower = 1.0
        }
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "scourge_rebels_demand"
	
	# Country scope
	demands_enforced_effect = {
		add_stability = -6
        add_prestige = -200
        add_adm_power = -999
        add_dip_power = -999
        add_mil_power = -999
	}
}
