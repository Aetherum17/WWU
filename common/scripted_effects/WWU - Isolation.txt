init_default_isolation_state = {
    # Titan Isolation
    if = {
        limit = {
            culture_group = culture_group_titan
        }
        
        add_country_modifier = {
            name = enforced_isolation
            duration = -1
        }
        
        # Add cooldown on start so player cannot immediately switch
        add_country_modifier = {
            name = isolation_cooldown
            duration = 3650
            hidden = yes
        }
    }
}
