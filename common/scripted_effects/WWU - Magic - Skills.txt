setup_spellcaster_history = {
    add_ruler_modifier = {
        name = spellcaster_capabilities
        duration = -1
    }
    
    set_variable = {
        which = ruler_magic_aptitude
        value = $APTITUDE$
    }
    if = {
        limit = {
            check_variable = {
                 which = ruler_magic_aptitude
                 value = 5
            }
        }
        set_variable = {
            which = ruler_magic_aptitude
            value = 5
        }  
    }
    
    set_variable = {
        which = ruler_magic_finesse
        value = $FINESSE$
    }
    if = {
        limit = {
            check_variable = {
                 which = ruler_magic_finesse
                 value = 5
            }
        }
        set_variable = {
            which = ruler_magic_finesse
            value = 5
        }  
    }
    
    set_variable = {
        which = ruler_magic_concentration
        value = $CONCENTRATION$
    }
    if = {
        limit = {
            check_variable = {
                 which = ruler_magic_concentration
                 value = 5
            }
        }
        set_variable = {
            which = ruler_magic_concentration
            value = 5
        }  
    }
    
    set_variable = {
        which = ruler_magic_mana
        value = $MANA$
    }
    if = {
        limit = {
            check_variable = {
                 which = ruler_magic_mana
                 value = 5
            }
        }
        set_variable = {
            which = ruler_magic_mana
            value = 5
        }  
    }
}

reset_spellcaster_skills = {
    set_variable = {
        which = ruler_magic_aptitude
        value = 0
    }
    set_variable = {
        which = ruler_magic_finesse
        value = 0
    }
    set_variable = {
        which = ruler_magic_concentration
        value = 0
    }
    set_variable = {
        which = ruler_magic_mana
        value = 0
    }
}

cataclysmic_effect = {
    if = {
        limit = {
            is_immortal_lifespan = yes
        }
        custom_tooltip = CATACLYSMIC_EFFECT_IMMORTAL_TT
    }
    else = {
        custom_tooltip = CATACLYSMIC_EFFECT_TT
    }
    
    
    hidden_effect = {
        random_list = {
            20 = {
                change_variable = {
                    which = ruler_magic_aptitude
                    value = -5
                }
                if = {
                    limit = {
                        NOT = {
                            check_variable = {
                                which = ruler_magic_aptitude
                                value = 0
                            }
                        }
                    }
                    set_variable = {
                        which = ruler_magic_aptitude
                        value = 0
                    }
                }
            }
            20 = {
                change_variable = {
                    which = ruler_magic_finesse
                    value = -5
                }
                if = {
                    limit = {
                        NOT = {
                            check_variable = {
                                which = ruler_magic_finesse
                                value = 0
                            }
                        }
                    }
                    set_variable = {
                        which = ruler_magic_finesse
                        value = 0
                    }
                }
            }
            20 = {
                change_variable = {
                    which = ruler_magic_concentration
                    value = -5
                }
                if = {
                    limit = {
                        NOT = {
                            check_variable = {
                                which = ruler_magic_concentration
                                value = 0
                            }
                        }
                    }
                    set_variable = {
                        which = ruler_magic_concentration
                        value = 0
                    }
                }
            }
            10 = {
                if = {
                    limit = {
                        is_immortal_lifespan = yes
                    }
                    country_event = { id = wwu_magic.22 }
                }
                else = {
                    country_event = { id = wwu_magic.21 }
                }
            }
            30 = {
                
            }
        }
    }
}
