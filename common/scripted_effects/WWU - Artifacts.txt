# ROOT = Losing Country, FROM = Winning Country
artifact_on_battle_lost_country_effect = {
    # Artifact: Strom'kar
    if = {
        limit = {
            # Loser has a general
            has_leaders = {
                value = 1
                type = general
                include_monarch = no
                include_heir = no
            }
            
            # Loser is troll
            culture_group = culture_group_troll
            
            # Winner has Strom'kar
            FROM = {
                custom_trigger_tooltip = {
                    tooltip = IS_PRIMARY_ARTIFACT_stromkar
                    has_country_flag = primary_artifact_stromkar
                }
            }
        }
        
        # Loser
        country_event = { id = wwu_archaeology.511 }
        
        # Winner event
        FROM = {
            country_event = { id = wwu_archaeology.510 }
        }
    }
}

start_dig = {
    hidden_effect = {
        set_country_flag = dig_begins
        set_country_flag = excavation_decision_used
        set_variable = {
            which = excavation_progress
            value = 0
        }
        set_variable = {
            which = excavation_progress_cap
            value = 8
        }
        
        # If a specific focus is chosen, make the dig harder
        if = {
            limit = {
                NOT = { has_country_flag = artifact_focus_random }
            }
            
            set_variable = {
                which = excavation_progress_cap
                value = 12
            }
        }
        
        # Conditional Modifiers
        if = {
            limit = { 
                has_country_flag = excavation_progress_boost
            }
            
            subtract_variable = {
                which = excavation_progress_cap
                value = 2
            }
        }
        if = {
            limit = { 
                has_country_modifier = dwarf_administration
            }
            
            subtract_variable = {
                which = excavation_progress_cap
                value = 2
            }
        }
        if = {
            limit = { 
                has_reform = wwu_labor_tier_archaeological_society
            }
            
            subtract_variable = {
                which = excavation_progress_cap
                value = 2
            }
        }
    }
}

increment_dig = {
    custom_tooltip = INCREMENT_DIG_TT
    
    hidden_effect = {
        change_variable = {
            which = excavation_progress
            value = 1
        }
        
        if = {
            limit = {
                check_variable = {
                    which = excavation_progress
                    which = excavation_progress_cap
                }
            }
            set_variable = {
                which = excavation_progress
                which = excavation_progress_cap
            }
        }
    }
}

increment_dig_massively = {
    custom_tooltip = INCREMENT_DIG_MASSIVELY_TT
    
    hidden_effect = {
        change_variable = {
            which = excavation_progress
            value = 3
        }
        
        if = {
            limit = {
                check_variable = {
                    which = excavation_progress
                    which = excavation_progress_cap
                }
            }
            set_variable = {
                which = excavation_progress
                which = excavation_progress_cap
            }
        }
    }
}

decrement_dig = {
    custom_tooltip = DECREMENT_DIG_TT
    
    hidden_effect = {
        subtract_variable = {
            which = excavation_progress
            value = 1
        }
 
        if = {
            limit = {
                NOT = { 
                    check_variable = {
                        which = excavation_progress
                        value = 0
                    }
                }
            }
            
            # End the dig if it is reduced to 0
            end_dig = yes
        }
    }
}

reset_dig = {
    custom_tooltip = RESET_DIG_TT
    
    hidden_effect = {
        set_variable = {
            which = excavation_progress
            value = 0
        }
    }
}

unlock_artifact = {
    custom_tooltip = UNLOCK_ARTIFACT_$artifact$
    
    hidden_effect = {
        set_country_flag = artifact_$artifact$
        set_global_flag = discovered_artifact_$artifact$
    }
}

remove_artifact = {
    hidden_effect = {
        clr_country_flag = artifact_$artifact$
        clr_global_flag = discovered_artifact_$artifact$
    }
}

make_primary_artifact = {
    custom_tooltip = MAKE_PRIMARY_ARTIFACT_$artifact$
    
    hidden_effect = {
        clear_current_primary_artifact = yes
        set_country_flag = primary_artifact_$artifact$
    }
}

steal_artifact = {
    custom_tooltip = STEAL_ARTIFACT_$artifact$
    
    hidden_effect = {
        set_country_flag = artifact_$artifact$
        
        event_target:artifact_heist_target = {
            set_country_flag = artifact_$artifact$_stolen
            clr_country_flag = artifact_$artifact$
            clr_country_flag = primary_artifact_$artifact$
        }
    }
}

display_artifact_status = {
    if = {
        limit = {
            any_country = {
                has_country_flag = artifact_$artifact$
            }
        }
        custom_tooltip = LOCATION_OF_$artifact$
    }
    else = {
        custom_tooltip = NOT_DISCOVERED_$artifact$
    }
}

save_artifact_location = {
    if = {
        limit = {
            has_global_flag = discovered_artifact_$artifact$
        }
        every_country = {
            limit = {
                 has_country_flag = artifact_$artifact$
            }
			# Audax Validator "." Ignore_NEXT
            save_event_target_as = current_artifact_owner
        }
    }
    else = {
        # Gaia represents no owner
        GAI = {
            save_event_target_as = current_artifact_owner
        }
    }
}
