increment_dig = {
    custom_tooltip = INCREMENT_DIG_TT
    
    hidden_effect = {
        change_variable = {
            which = excavation_progress
            value = 1
        }
        
        if = {
            limit = { 
                has_country_flag = excavation_progress_boost
            }
            
            change_variable = {
                which = excavation_progress
                value = 1
            }
        }
        
        if = {
            limit = {
                check_variable = {
                    which = excavation_progress
                    value = 5
                }
            }
            set_variable = {
                which = excavation_progress
                value = 5
            }
        }
    }
}

decrement_dig = {
    custom_tooltip = DECREMENT_DIG_TT
    
    hidden_effect = {
        subtract_variable = {
            which = excavation_progress
            value = 1
        }
        
        if = {
            limit = {
                NOT = { 
                    check_variable = {
                        which = excavation_progress
                        value = 0
                    }
                }
            }
            set_variable = {
                which = excavation_progress
                value = 0
            }
        }
    }
}

unlock_artifact = {
    custom_tooltip = UNLOCK_ARTIFACT_$artifact$
    
    hidden_effect = {
        set_country_flag = artifact_$artifact$
        set_global_flag = discovered_artifact_$artifact$
    }
}

remove_artifact = {
    hidden_effect = {
        clr_country_flag = artifact_$artifact$
        clr_global_flag = discovered_artifact_$artifact$
    }
}

make_primary_artifact = {
    custom_tooltip = MAKE_PRIMARY_ARTIFACT_$artifact$
    
    hidden_effect = {
        clear_current_primary_artifact = yes
        set_country_flag = primary_artifact_$artifact$
    }
}

steal_artifact = {
    custom_tooltip = STEAL_ARTIFACT_$artifact$
    
    hidden_effect = {
        set_country_flag = artifact_$artifact$
        
        event_target:artifact_heist_target = {
            set_country_flag = artifact_$artifact$_stolen
            clr_country_flag = artifact_$artifact$
            clr_country_flag = primary_artifact_$artifact$
        }
    }
}

display_artifact_status = {
    if = {
        limit = {
            any_country = {
                has_country_flag = artifact_$artifact$
            }
        }
        custom_tooltip = LOCATION_OF_$artifact$
    }
    else = {
        custom_tooltip = NOT_DISCOVERED_$artifact$
    }
}

save_artifact_location = {
    if = {
        limit = {
            has_global_flag = discovered_artifact_$artifact$
        }
        every_country = {
            limit = {
                 has_country_flag = artifact_$artifact$
            }
            save_event_target_as = current_artifact_owner
        }
    }
    else = {
        # Gaia represents no owner
        GAI = {
            save_event_target_as = current_artifact_owner
        }
    }
}
