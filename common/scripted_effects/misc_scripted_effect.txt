# Adds two variables and writes output to first value.
# USE:  add_variables = { var1 = var_x var2 = var_y }
add_variables = {
	set_variable = {
		which = counter1
		value = 1
	}
	while = {
		limit = {
			check_variable = {
				which = $var2$
				which = counter1
			}
		}
		change_variable = {
			which = counter1
			value = 1
		}
		change_variable = {
			which = $var1$
			value = 1
		}
	}
	set_variable = {
		which = counter1
		value = 0
	}
}
# SuÄ±btracts second value from the first and writes output to first value.
# USE:  sub_variables = { var1 = var_x var2 = var_y }
sub_variables = {
	set_variable = {
		which = counter1
		value = 1
	}
	while = {
		limit = {
			check_variable = {
				which = $var2$
				which = counter1
			}
		}
		change_variable = {
			which = counter1
			value = 1
		}
		change_variable = {
			which = $var1$
			value = -1
		}
	}
	set_variable = {
		which = counter1
		value = 0
	}
}

# Multiplies two variables and writes output to first value.
# USE:  multiply_variables = { var1 = 3 var2 = 2 }
# NOTE: Do not use this with too large values, It will cause game to freeze.
multiply_variables = {
	set_variable = {
		which = counter2
		value = 2
	}
	set_variable = {
		which = var1_temp
		which = $var1$
	}
	# Operating of a value zero times always return that operation's identity value.
	if = {
		limit = {
			is_variable_equal = {
				which = $var2$
				value = 0
			}
		}
		set_variable = {
			which = $var1$
			value = 0
		}
	}
	else = {
		while = {
			limit = {
				check_variable = {
					which = $var2$
					which = counter2
				}
			}
			change_variable = {
				which = counter2
				value = 1
			}
			add_variables = {
				var1 = $var1$
				var2 = var1_temp
			}
		}
		set_variable = {
			which = counter2
			value = 0
		}
		set_variable = {
			which = var1_temp
			value = 0
		}
	}
}
