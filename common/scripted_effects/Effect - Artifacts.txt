on_yearly_pulse__artifacts = {
    # Clean up Artifact Heist if the normal cleanup event does not occur.
    if = {
        limit = {
            had_country_flag = {
                flag = artifact_heist_active
                days = 1800
            }
        }
        
        clr_country_flag = artifact_heist_active
    }
}

on_five_year_pulse_2__artifacts = {
    # Pearl of Pandaria
    if = {
        limit = {
            has_country_flag = pearl_of_pandaria_search
            NOT = { has_global_flag = discovered_artifact_pearl_of_pandaria }
        }
        
        random_list = {
            10 = { country_event = { id = wwu_archaeology.85 } }
            90 = { }
        }
    }
}

on_battle_lost_country__artifacts = {
    # Artifact: Strom'kar
    if = {
        limit = {
            # Loser has a general
            has_leaders = {
                value = 1
                type = general
                include_monarch = no
                include_heir = no
            }
            
            # Loser is troll
            culture_group = culture_group_troll
            
            # Winner has Strom'kar
            FROM = {
                custom_trigger_tooltip = {
                    tooltip = IS_PRIMARY_ARTIFACT_stromkar
                    has_country_flag = primary_artifact_stromkar
                }
            }
        }
        
        # Loser
        country_event = { id = wwu_archaeology.511 }
        
        # Winner event
        FROM = {
            country_event = { id = wwu_archaeology.510 }
        }
    }
}

    hidden_effect = {

        set_country_flag = excavation_decision_used
        
        set_variable = {
            which = excavation_progress
            value = 0
        }

monthly_dig_progression = {
    if = {
        limit = {
            has_country_flag = dig_begins
        }
        
        # If a specific focus is chosen, apply a slower gain
        if = {
            limit = {
                NOT = { has_country_flag = artifact_focus_random }
            }
            
            random_list = {
                20 = { }
                20 = { change_variable = { which = excavation_progress value = 0.1 } }
                20 = { change_variable = { which = excavation_progress value = 0.2 } }
                20 = { change_variable = { which = excavation_progress value = 0.3 } }
                20 = { change_variable = { which = excavation_progress value = 0.4 } }
                20 = { change_variable = { which = excavation_progress value = 0.5 } }
            }
        }
        else = {
            random_list = {
                20 = { change_variable = { which = excavation_progress value = 0.5 } }
                20 = { change_variable = { which = excavation_progress value = 0.6 } }
                20 = { change_variable = { which = excavation_progress value = 0.7 } }
                20 = { change_variable = { which = excavation_progress value = 0.8 } }
                20 = { change_variable = { which = excavation_progress value = 0.9 } }
                20 = { change_variable = { which = excavation_progress value = 1.0 } }
            }
        }

        # Conditional Modifiers
        if = {
            limit = { 
                has_country_flag = excavation_progress_boost
            }
            
            change_variable = { which = excavation_progress value = 0.5 }
        }
        if = {
            limit = { 
                has_country_modifier = dwarf_administration
            }
            
            change_variable = { which = excavation_progress value = 0.5 }
        }
        if = {
            limit = { 
                has_reform = wwu_labor_tier_archaeological_society
            }
            
            change_variable = { which = excavation_progress value = 0.5 }
        }
    }
}
