# Spawn Freely
spawn_nation_by_province = {
    $province$ = {
        cede_province = $tag$
        add_core = $tag$
        change_culture = $culture$
        change_religion = $religion$
    }
}

spawn_nation_by_area = {
    if = {
        limit = {
            $area$ = {
                NOT = { owned_by = ROOT } # This is so nations other than ROOT don't suddenly lose land
            }   
        }
        $area$ = {
            limit = {
                NOT = { owned_by = ROOT }
            }
            add_core = $tag$
        }
    }
    if = {
        limit = {
            $area$ = {
                owned_by = ROOT
            }   
        }
        $area$ = {
            limit = {
                owned_by = ROOT
            }
            cede_province = $tag$
            add_core = $tag$
            change_culture = $culture$
            change_religion = $religion$
        }
    }
}

spawn_nation_by_flag = {
    if = {
        limit = {
            any_province = {
                has_province_flag = $flag$
                NOT = { owned_by = ROOT } # This is so nations other than ROOT don't suddenly lose land
            }
        }
        every_province = {
            limit = {
                has_province_flag = $flag$
                NOT = { owned_by = ROOT }
            }
            add_core = $tag$
        }
    }
    if = {
        limit = {
            any_owned_province = {
                has_province_flag = $flag$
            }            
        }
        every_owned_province = {
            limit = {
                has_province_flag = $flag$
            }
            cede_province = $tag$
            add_core = $tag$
            change_culture = $culture$
            change_religion = $religion$
        }
    }
}

# Spawn Conditionally
spawn_nation_by_province_conditional = {
    if = {
        limit = {
            $province$ = {
                is_empty = yes
            }
        }
        
        $province$ = {
            cede_province = $tag$
            add_core = $tag$
            change_culture = $culture$
            change_religion = $religion$
        }
    }
    else_if = {
        limit = {
            $province$ = {
                culture = ROOT
            }
        }
        
        $province$ = {
            add_core = $tag$
        }
    }
    else = {
        $province$ = {
            add_core = $tag$
            change_culture = $culture$
            change_religion = $religion$
        }
    }
}

# Spawn Cores Only

        add_core = $tag$
    }
}

    }
}

        }

    }
}

# Flags
clear_province_flags = {
    every_province = {
        limit = {
            has_province_flag = $flag$
        }
        
        clr_province_flag = $flag$
    }
}
