#-----------------------------------
# On Actions
#-----------------------------------
on_battle_won_country__statistics = {
	if = {
		limit = {
			FROM = { tag = REB }
		}
		change_variable = {
			which = rebels_suppressed
			value = 1
		}
	}
}

    change_variable = {
		which = war_lost
		value = 1
	}
}

        value = 1
    }
}
