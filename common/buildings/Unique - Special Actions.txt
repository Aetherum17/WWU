province_action_tax = {
	cost = 1000
	time = 64
    
    build_trigger = {
        owner = {
            NOT = { has_country_modifier = building_tax_development_cooldown }
        }
    }
    
    on_built = {
        log = "DEBUG - [Root.GetName] - province_action_tax"
        
        add_base_tax = 1
        remove_building = province_action_tax
        
        owner = {
            add_country_modifier = {
                name = building_tax_development_cooldown
                duration = 1095
            }
        }
    }
	
	ai_will_do = {
		factor = 10.0

        modifier = {
            factor = 0
            
            base_tax = 6
        }

		modifier = {
			factor = 0
			FROM = { num_of_loans = 3 }
		}
		modifier = {
			factor = 0.3
			FROM = { num_of_loans = 2 }
		}
		modifier = {
			factor = 0.5
			FROM = { num_of_loans = 1 }
		}
		modifier = {
			factor = 0.75
			FROM = { NOT = { treasury = 10000 } }
		}
		modifier = {
			factor = 0.75
			FROM = { is_in_deficit = yes }
		}
    }
}
