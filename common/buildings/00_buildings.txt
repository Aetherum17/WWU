wow_fort_1 = {
	cost = 100
	time = 30
    
	modifier = {
		fort_level = 2
        garrison_growth = 0.2
	}
	
	onmap = yes
	
	influencing_fort = yes
	
	ai_will_do = {
		factor = 1000
		modifier = {
			factor = 0.5
			NOT = {
				any_neighbor_province = {
					NOT = { owned_by = ROOT }
				}
			}
		}			
		modifier = {
			factor = 2.0
			any_neighbor_province = {
				NOT = { owned_by = ROOT }
			}
		}	
		modifier = {
			factor = 0.1
			is_overseas = yes
		}	
		modifier = {
			factor = 0
			any_neighbor_province = {
				owned_by = ROOT
				OR = {
					has_building = wow_fort_1
					has_building = wow_fort_2
					has_building = wow_fort_3
					has_building = wow_fort_4
				}
			}
		}		
	}
}

wow_garrison = {
	manufactory = {
		all
	}

	build_trigger = {
		NOT = {
			area_for_scope_province = {
				owner = {
					tag = ROOT
				}
				OR = {
					has_building = wow_garrison
					has_construction = wow_garrison
				}
			}
		}
	}
	
	keep_trigger = {
		NOT = {
			area_for_scope_province = {
				NOT = { province_id = root }
				owner = {
					tag = ROOT
				}
				OR = {
					has_building = wow_garrison
					has_construction = wow_garrison
				}
			}
		}
	}

	modifier = {
		statewide_governing_cost = -0.5
		min_local_autonomy = -5
	}
    
	show_separate = yes 

	ai_will_do = {

        modifier = {
			factor = 2
            owner = { num_of_states = 25 }
		}
        modifier = {
			factor = 2
            owner = { num_of_states = 50 }
		}
