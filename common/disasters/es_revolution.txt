es_revolution = {
	potential = {
		NOT = { has_country_modifier = complete_stasis }
		NOT = { 
			capital_scope = {
				continent = continent_pandaria
			}
		}
	}

	can_start = {
		OR = {
			AND = {
				NOT = { legitimacy_equivalent = 25 }
				any_subject_country = {
					is_subject_of_type = faction_alliance
					liberty_desire = 75
				}
			}
			AND = {
				has_country_modifier = is_evil
				any_subject_country = {
					is_subject_of_type = faction_horde
					NOT = { has_country_modifier = is_evil }
					liberty_desire = 75
				}
			}
			AND = {
				NOT = { legitimacy_equivalent = 25 }
				is_civilized = yes
				has_country_modifier = is_evil
			}
		}
		is_lesser_in_union = no
		is_free_or_tributary_trigger = yes
	}
	
	
	can_stop = {
		OR = {
			any_subject_country = {
				is_subject_of_type = faction_horde
				NOT = { liberty_desire = 25 }
			}
			NOT = { has_country_modifier = is_evil }
			legitimacy_equivalent = 75
		}
	}
		
	progress = {
	    modifier = {
			factor = 1
			legitimacy_equivalent = 75
		}
		modifier = {
			factor = 1
			revolt_percentage = 0.10
		}	
		modifier = {
			factor = 1
			overextension_percentage = 0.10
		}
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
        modifier = {
			factor = 1
			war_exhaustion = 5
		} 	
	}
	
	can_end = {
	    legitimacy_equivalent = 90
	}

	modifier = {
	    stability_cost_modifier = 1.0
		global_unrest = 10
	}	
	
	
	on_start = ge_revolution.1
	on_end = ge_revolution.27
	
	on_monthly = {
		events = {
		}
	}
}
