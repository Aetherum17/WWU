# Traders
estate_monopoly_of_resource_common_wood = {
	icon = privilege_resource_common_wood
	max_absolutism = -10

	loyalty = 0.05
	is_valid = {
		any_owned_province = {
			trade_goods = resource_common_wood
		}
	}
	can_select = {
	}
	on_granted = {
		if = {
			limit = {
				resource_common_wood = 1
			}
			add_mercantilism = 1
		}
		add_years_of_owned_provinces_production_income = {
			years = 8
			trigger = {
				trade_goods = resource_common_wood
			}
			custom_tooltip = MONOPOLY_INCOME_RESOURCE_COMMON_WOOD_DESC
		}
		custom_tooltip = estate_monopoly_of_resource_common_wood_effect_tooltip
	}
	on_granted_province = {
		if = {
			limit = {
				trade_goods = resource_common_wood
			}
			add_province_modifier = {
				name = estate_monopoly_of_resource_common_wood_mod
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = estate_monopoly_of_resource_common_wood_mod
			}
			remove_province_modifier = estate_monopoly_of_resource_common_wood_mod
		}
	}
	on_revoked = {
		custom_tooltip = estate_monopoly_of_resource_common_wood_revoke_tooltip
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = estate_monopoly_of_resource_common_wood_mod
			}
			remove_province_modifier = estate_monopoly_of_resource_common_wood_mod
		}
	}
	on_invalid = {
		custom_tooltip = estate_monopoly_of_resource_common_wood_revoke_tooltip
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = estate_monopoly_of_resource_common_wood_mod
			}
			remove_province_modifier = estate_monopoly_of_resource_common_wood_mod
		}
	}
	penalties = {
	}
	benefits = {
	}

	cooldown_years = 10

	on_cooldown_expires = {
		country_event = { id = wwu_monopoly_privileges.1 }
	}
	ai_will_do = {
		factor = 0
	}
}

estate_monopoly_of_resource_fel = {
	icon = privilege_resource_fel
	max_absolutism = -10

	loyalty = 0.05
	is_valid = {
		any_owned_province = {
			trade_goods = resource_fel
		}
	}
	can_select = {
	}
	on_granted = {
		if = {
			limit = {
				resource_fel = 1
			}
			add_mercantilism = 1
		}
		add_years_of_owned_provinces_production_income = {
			years = 8
			trigger = {
				trade_goods = resource_fel
			}
			custom_tooltip = MONOPOLY_INCOME_RESOURCE_FEL_DESC
		}
		custom_tooltip = estate_monopoly_of_resource_fel_effect_tooltip
	}
	on_granted_province = {
		if = {
			limit = {
				trade_goods = resource_fel
			}
			add_province_modifier = {
				name = estate_monopoly_of_resource_fel_mod
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = estate_monopoly_of_resource_fel_mod
			}
			remove_province_modifier = estate_monopoly_of_resource_fel_mod
		}
	}
	on_revoked = {
		custom_tooltip = estate_monopoly_of_resource_fel_revoke_tooltip
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = estate_monopoly_of_resource_fel_mod
			}
			remove_province_modifier = estate_monopoly_of_resource_fel_mod
		}
	}
	on_invalid = {
		custom_tooltip = estate_monopoly_of_resource_fel_revoke_tooltip
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = estate_monopoly_of_resource_fel_mod
			}
			remove_province_modifier = estate_monopoly_of_resource_fel_mod
		}
	}
	penalties = {
	}
	benefits = {
	}

	cooldown_years = 10

	on_cooldown_expires = {
		country_event = { id = wwu_monopoly_privileges.42 }
	}
	ai_will_do = {
		factor = 0
	}
}
