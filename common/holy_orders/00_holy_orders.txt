# Class order halls from legion xpac

warrior_order = {
	icon = GFX_class_warrior
	trigger = { NOT = { has_country_flag = has_warrior_order } NOT = { any_country = { has_country_flag = has_warrior_order } } }
	
    cost = 150
    cost_type = mil_power
	
    per_province_effect = {
        add_local_autonomy = 10
		add_prestige = 1
		owner = { set_country_flag = has_warrior_order }
    }
    per_province_abandon_effect = {
	    add_local_autonomy = -10
		add_prestige = -1
		owner = { clr_country_flag = has_warrior_order }
    }
    modifier = {
        garrison_growth = 0.1
    }
    ai_priority = {
        factor = 1
    }
}
