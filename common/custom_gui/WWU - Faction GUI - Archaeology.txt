# Actions
custom_button = {
    name = archaeology_gui_start_dig_button
    tooltip = ARCHAEOLOGY_START_DIG_TT
    
    potential = {
        is_archaeology_gui_open = yes
        
    }
    
    trigger = { 
        custom_trigger_tooltip = {
            tooltip = ARCHAEOLOGY_NOT_STARTED_DIG_TT
            
            NOT = { has_country_flag = excavation_decision_used }
        }
        
        custom_trigger_tooltip = {
            tooltip = ARCHAEOLOGY_NOT_USED_DIG_RECENTLY_TT
            
            NOT = { has_country_modifier = archaeology_dig_decision_cooldown }
        }
        
        is_allowed_to_dig = yes
        
        custom_trigger_tooltip = {
            tooltip = HAS_NOT_STARTED_ARCHAEOLOGICAL_DIG
            NOT = { has_country_flag = excavation_decision_used }
        }
        
        adm_power = 50
        dip_power = 50
        mil_power = 50
    }
    
    effect = { 
        add_adm_power = -50
        add_dip_power = -50
        add_mil_power = -50
            
        hidden_effect = {
            set_country_flag = excavation_decision_used
            
            add_country_modifier = {
                name = archaeology_dig_decision_cooldown
                duration = 1800
                hidden = yes
            }
            
            set_variable = {
                which = excavation_progress
                value = 0
            }
        }
        
        country_event = { id = wwu_archaeology.300 }
    }
}

    name = archaeology_gui_boost_dig_button
    tooltip = ARCHAEOLOGY_BOOST_DIG_TT
    
    potential = {
        has_country_flag = dig_begins
        is_archaeology_gui_open = yes
    }
    
    trigger = {
        custom_trigger_tooltip = {
            tooltip = ARCHAEOLOGY_NOT_USED_DIG_BOOST_RECENTLY_TT
            
            NOT = { has_country_modifier = archaeology_dig_status_action_cooldown }
        }
        years_of_income = 1.0
    }
    
    effect = {
        add_years_of_income = -1.0
        advance_dig_action = yes
        
        custom_tooltip = ARCHAEOLOGICAL_EXPEDITION_ACTION_COOLDOWN_TT
        hidden_effect = {
            add_country_modifier = {
                name = archaeology_dig_status_action_cooldown
                duration = 365
                hidden = yes
            }
        }
    }
}
