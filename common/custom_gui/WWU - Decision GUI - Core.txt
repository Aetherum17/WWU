custom_button = {
    name = enable_major_decisions_button
    tooltip = enable_major_decisions_tt
    
    potential = {
        NOT = { has_country_flag = display_major_decisions }
    }
    
    trigger = { 
    
    }
    
    effect = { 
        custom_tooltip = SHOW_MAJOR_DECISIONS
        hidden_effect = {
            set_country_flag = display_major_decisions
        }
    }
}
custom_button = {
    name = disable_major_decisions_button

    }
    
    effect = { 
        custom_tooltip = HIDE_MAJOR_DECISIONS
        hidden_effect = {
            clr_country_flag = display_major_decisions
        }
    }
}
