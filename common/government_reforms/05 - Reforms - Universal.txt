# Labor Organisation (Level 1)
#-----------------------------------------
wwu_labor_tier_commission = {
	icon = "wwu_labor_tier_commission"
	allow_normal_conversion = yes

    trigger = {
        NOT = {
            estate_influence = { 
                estate = estate_slaves
                influence = 5
            }
        }
    }

	modifiers = {
		build_cost = -0.1
        global_tax_modifier = 0.1
	}
}
wwu_labor_tier_guild_labor = {
	icon = "wwu_labor_tier_guild_labor"
	allow_normal_conversion = yes

    trigger = {
        NOT = {
            estate_influence = { 
                estate = estate_slaves
                influence = 5
            }
        }
    }
    
	modifiers = {
		build_time = -0.3
        production_efficiency = 0.05
	}
}
wwu_labor_tier_serfdom = {
	icon = "wwu_labor_tier_serfdom"
	allow_normal_conversion = yes

    trigger = {
        has_estate = estate_peasants
        NOT = {
            estate_influence = { 
                estate = estate_slaves
                influence = 5
            }
        }
    }

	modifiers = {
		state_governing_cost = -0.15
	}
    
    effect = {
        set_estate_privilege = estate_peasants_serfdom_enforced
    }
    
    removed_effect = {
        remove_estate_privilege = estate_peasants_serfdom_enforced
    }
}
wwu_labor_tier_slavery = {
	icon = "wwu_labor_tier_slavery"
	allow_normal_conversion = yes

    trigger = {
        OR = {
            has_estate = estate_peasants
            has_estate = estate_slaves
        }
    }

	modifiers = {
		global_manpower_modifier = 0.05
        diplomatic_reputation = -2
	}
    
    effect = {
        custom_tooltip = enable_slave_estate_tt
    }
    removed_effect = {
        custom_tooltip = disable_slave_estate_tt
    }
}
