# Katrana Prestor
unique_katrana_prestor = {
	monarch_power = ADM
	
	# Bonus
	yearly_corruption = -0.05

	skill_scaled_modifier = { 
        trigger = {
            owner = {
                uses_meritocracy = yes
            }
        }
        
        modifier = {
			meritocracy = 0.5 
		}
    }

	chance = {
		factor = 100
        
        modifier = {
            factor = 0
            
            NOT = { current_age = age_trouble_in_azeroth }
        }
        
        # Is past life span of advisor
        modifier = {
            factor = 0
            
            NOT = { is_year = 700 }
        }
        
        # Is not Lordaeron
        modifier = {
            factor = 0
            
            owner = { 
                NOT = { tag = A08 }
            }
        }
	}
	
	ai_will_do = {
		factor = 10
	}
}
