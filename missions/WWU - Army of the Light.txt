ARL_army_of_the_light_column_4 = {
    slot = 4
	generic = no
	ai = yes
    has_country_shield = yes

    potential = {
        tag = ARL
    }

    ARL_contact_the_draenei = {
        icon = mission_titan_spirits
        position = 2

        trigger = {
            if = {
                limit = {
                    OR = {
                        exists = A54
                        exists = A56
                    }
                }
                OR = {
                    has_opinion = {
                        who = A54 # Shattarah
                        value = 100
                    }
                    has_opinion = {
                        who = A56 # Exodar
                        value = 100
                    }
                }
            }
            else = {
                stability = 1
            }
        }

        effect = {
            if = {
                limit = {
                    exists = A54
                }
                create_alliance = A54
                add_historical_friend = A54
                A54 = {
                    add_historical_friend = ROOT
                    add_accepted_culture = culture_lightforged
                }
            }
            else = {
                add_mil_power = 100
            }
            
            if = {
                limit = {
                    exists = A56
                }
                # Exodar
                create_alliance = A56
                add_historical_friend = A56
                A56 = {
                    add_historical_friend = ROOT
                    add_accepted_culture = culture_lightforged
                }
            }
            else = {
                add_mil_power = 100
            }
            add_accepted_culture = culture_draenei
        }
    }

    ARL_joining_the_alliance = {
        icon = mission_titan_spirits
        position = 2

        required_missions = {}

        trigger = {
                any_country = {
                    has_country_modifier = head_of_alliance
                    has_opinion = {
                        who = ROOT
                        value = 120
                    }
                    alliance_with = ROOT
            }
        }

        provinces_to_highlight = {
            is_capital = yes
            owner = {
                has_country_modifier = head_of_alliance
            }
        }

        effect = {
            add_country_modifier = {
                name = member_of_alliance
                duration = -1
            }

            add_adm_power = 100
            add_dip_power = 100
            add_mil_power = 100
        }
    }

    ARL_purification_of_krokul = {
        icon = mission_titan_spirits
        position = 4

        required_missions = { ARL_securing_krokul }

        trigger = {
            region_krokuul = {
                type = all
                OR = {
                    culture = culture_lightforged
                    culture = culture_broken
                }
            }
        }

        provinces_to_highlight = {
            region = region_krokuul
        }

        effect = {
            add_country_modifier = {
                name = mission_unity_of_krokul
                duration = 3650
            }
            add_mil_power = 200
        }
    }       
}
