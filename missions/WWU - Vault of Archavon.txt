VAV_titan_column_1_1 = {
    slot = 1
    generic = no
    ai = yes
 
    potential = {
        tag = VAV
        has_personal_tab_active = yes
            NOT = {
                check_variable = {
                    which = archavon_trap_2_reward
                    value = 1
                }
            }
    }
    has_country_shield = yes
 
    # 2 out of 20 mission for second trap
    VAV_one_two_1 = {
        icon = mission_nerub_vrykul
        position = 2
        required_missions = { }
 
        trigger = {
			custom_trigger_tooltip = {
				tooltip = archavon_mission_one_two_trigger_1_tt
                check_variable = {
                    which = archavon_trap_2_extra
                    value = 1
                }
			}
        }
 		effect = {
            add_adm_power = 10
            add_dip_power = 10
            add_mil_power = 10
            if = {
                limit = {
                    NOT = {
                        is_subject = yes
                    }
                }
                add_splendor = 10
                tooltip = archavon_mission_overlord_1_tt
            }
            if = {
                limit = {
                    is_subject = yes
                }
                tooltip = archavon_mission_overlord_2_tt
                overlord = {
                    add_adm_power = 10
                    add_dip_power = 10
                    add_mil_power = 10
                }
            }
            tooltip = archavon_mission_titans_facility_add_1_tt
            hidden_effect = { swap_non_generic_missions = yes }
            hidden_effect = {
                change_variable = {
                    which = archavon_trap_2_reward
                    value = 1
                }
                change_variable = {
                    which = archavon_titans_facility
                    value = 1
                }
            }
        }
 
        ai_weight   = { factor = 1 }
    }
}
