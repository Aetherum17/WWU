# Apocalypse
country_event = {
	id = wwu_artifact_repository.47
	title = wwu_artifact_repository.47.title
	desc = wwu_artifact_repository.47.desc
	picture = TREASURE_VAULT_eventPicture

	is_triggered_only = yes

	immediate = {
        hidden_effect = {
            save_artifact_location = { artifact = apocalypse }
        }
    }

	option = {
		name = wwu_artifact_repository.47.option.make_primary

        trigger = {
            has_country_flag = artifact_apocalypse
        }

        make_primary_artifact = {
            artifact = apocalypse
        }

        hidden_effect = {
            country_event = { id = wwu_artifact_repository.1 }

            if = {
                limit = {
                    ai = yes
                }
                add_country_modifier = {
                    name = ai_recently_set_primary_artifact
                    duration = 9125
                }
            }
        }

        ai_chance = {
            factor = 100
        }
	}
    option = {
		name = wwu_artifact_repository.47.option.exit

        hidden_effect = {
            country_event = { id = wwu_artifact_repository.1 }
        }
	}
}
