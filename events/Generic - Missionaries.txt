	title = wwu_missionaries.100.title
	desc = wwu_missionaries.100.desc
	picture = JUNGLE_GIANTS_eventPicture
    
    is_triggered_only = yes
    
	trigger = {
        any_owned_province = {
            has_missionary = yes
        }
	}
	
	mean_time_to_happen = {
		days = 1
	}
    
    immediate = {
        hidden_effect = {
            random_owned_province = {
                limit = { has_missionary = yes }
                save_event_target_as = target_province
            }
        }
    }
    
	option = {
		name = wwu_missionaries.1.option.a
        
        event_target:target_province = {
            random_list = {
                10 = { province_event = { id = wwu_missionaries.1 } }

# Missionary Falls Ill
province_event = {
	id = wwu_missionaries.1
	title = wwu_missionaries.1.title
	desc = wwu_missionaries.1.desc
	picture = SHADOW_MADNESS_eventPicture
    goto = ROOT
    
    is_triggered_only = yes
    
	trigger = {
		has_missionary = yes
		NOT = { has_building = wow_temple_1 }
		NOT = { has_province_modifier = missionary_falls_ill }
		owner = { adm_power = 25 }
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = wwu_missionaries.1.option.a
		
		ai_chance = { factor = 65 }
		owner = {
			add_adm_power = -25
			add_piety = 0.05
		}
	}
	option = {
		name = wwu_missionaries.1.option.b
		
		ai_chance = { factor = 35 }
		add_province_modifier = {
			name = "missionary_falls_ill"
			duration = 1825
		}
		owner = { add_piety = -0.05 }
	}
}
