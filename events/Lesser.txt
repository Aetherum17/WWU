country_event = {
	id = lesser.2
	title = axis_of_awful_offer.title #
	desc = axis_of_awful_offer.desc #
	picture = ELF_TROLL_eventPicture
	
	mean_time_to_happen = {
		months = 120
	}
	
	trigger = {
		is_lesser = yes
		NOT = { is_at_war = yes }
		any_country = {
			is_lesser = yes
			NOT = { tag = ROOT }
			NOT = {
				border_distance = {
					who = ROOT
					distance = 75
				}
			}
			has_opinion = {
				who = ROOT
				value = 0				
			}
		}
		any_country = {
			NOT = { coalition_target = THIS }
			NOT = { is_lesser = yes }
			NOT = { tag = ROOT }
			NOT = {
				border_distance = {
					who = ROOT
					distance = 75
				}
			}
			NOT = { 
				has_opinion = {
					who = ROOT
					value = 0				
				} 
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_country = {
				limit = {
					is_lesser = yes
					NOT = { tag = ROOT }
					NOT = {
						border_distance = {
							who = ROOT
							distance = 75
						}
					}
					has_opinion = {
						who = ROOT
						value = 0				
					}
				}
				save_event_target_as = potential_ally
			}
			random_country = {
				limit = {
					NOT = { coalition_target = THIS }
					NOT = { is_lesser = yes }
					NOT = { tag = ROOT }
					NOT = {
						border_distance = {
							who = ROOT
							distance = 75
						}
					}
					NOT = { 
						has_opinion = {
							who = ROOT
							value = 0				
						} 
					}
				}
				save_global_event_target_as = potential_target
			}
		}
	}
	option = {
		name = axis_of_awful_offer.opt1
		event_target:potential_ally = {
			country_event = { id = lesser.3 }
		}
	}
	option = {
		name = axis_of_awful_offer.opt2
		add_prestige = -5
	}
}
