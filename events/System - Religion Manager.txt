namespace = wwu_religion_manager

# Religious Management
country_event = {
	id = wwu_religion_manager.1
	title = wwu_religion_manager.1.title
	desc = wwu_religion_manager.1.desc
	picture = ELUNES_GRACE_eventPicture
	
	is_triggered_only = yes
    
    # Enable Automatic Religious Conversion
	option = {
		name = wwu_religion_manager.1.option.auto_convert.on
        
        trigger = {
            NOT = { has_country_flag = enabled_auto_religion_conversion }
        }
        
        custom_tooltip = AUTO_RELIGION_CONVERT_ON_TT
        hidden_effect = {
            set_country_flag = enabled_auto_religion_conversion
        }
	}
    
    # Disable Automatic Religious Conversion
	option = {
		name = wwu_religion_manager.1.option.auto_convert.off
        
        trigger = {
            has_country_flag = enabled_auto_religion_conversion
        }
        
        custom_tooltip = AUTO_RELIGION_CONVERT_OFF_TT
        hidden_effect = {
            clr_country_flag = enabled_auto_religion_conversion
        }
	}
    
    # Exit
	option = {
		name = wwu_culture_manager.1.option.exit
	}
    
    after = {
        if = {
            limit = {
                has_country_flag = skip_status_report_return
            }
            
            clr_country_flag = skip_status_report_return
        }
        else = {
            country_event = { id = wwu_status_report.2 }
        }
    }
}
