# Fate of Suramar
country_event = {
	id = wwu_nazjatar.3
	title = wwu_nazjatar.3.title
	desc = wwu_nazjatar.3.desc
	picture = SURAMAR_GARDEN_eventPicture

    is_triggered_only = yes
    
	mean_time_to_happen = { 
        days = 1
    }
    
	option = {
		name = wwu_nazjatar.3.option.a
        
        # This is to prevent the player from peacing out and then getting the event effect as a bonus.
        if = {
            limit = {
                war_with = SUR
            }
            
            hidden_effect = {
                white_peace = SUR
            }
            
            inherit = SUR
            
            custom_tooltip = DROWNED_PROVINCE_TT
            
			region_suramar = {
                add_permanent_province_modifier = {
                    name = mission_drowned_province
                    duration = -1
                }
            }
            
            every_country = {
                limit = {
                    capital_scope = {
                        superregion = broken_isles_superregion
                    }
                }
                
                add_aggressive_expansion = {
                    who = ROOT
                    value = 50
                    apply_calc = yes
                }
            }
        }
    }
    option = {
		name = wwu_nazjatar.3.option.b
        
        # This is to prevent the player from peacing out and then getting the event effect as a bonus.
        if = {
            limit = {
                war_with = SUR
            }
            
            hidden_effect = {
                white_peace = SUR
            }
            
            create_subject = {
                subject_type = vassal
                subject = SUR
            }
            
            every_country = {
                limit = {
                    capital_scope = {
                        superregion = broken_isles_superregion
                    }
                }
                
                add_aggressive_expansion = {
                    who = ROOT
                    value = 30
                    apply_calc = yes
                }
            }
        }
    }
    
}
