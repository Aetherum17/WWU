namespace = wwu_setup_pirate

# Bloodsail Buccaneers
country_event = {
	id = wwu_setup_pirate.1
	title = wwu_setup_pirate.1.title
	desc = wwu_setup_pirate.1.desc
	picture = BLOODSAIL_RAIDER_eventPicture

	is_triggered_only = yes
    
	trigger = {
        NOT = { exists = A15 }
        NOT = { has_global_flag = setup_bloodsail }
        exists = P11 # Scourge exists
        exists = A14 # Kul Tiras exists
        capital_scope = {
            superregion = plunder_isle_superregion
        }
    }
    
    immediate = {
        hidden_effect = {
            set_global_flag = setup_bloodsail
        }
    }
    
    mean_time_to_happen = {
		days = 1
	}

	option = {
		name = wwu_setup_pirate.1.option.a
        
        area_bloodsail_hold = {
            cede_province = A15
            add_core = A15
            change_culture = culture_kul_tiran
            change_religion = shadow
        }
        
        ai_chance = {
            factor = 90
        }
	}
	option = {
		name = wwu_setup_pirate.1.option.b
        
        trigger = {
            area_bloodsail_hold = {
                owned_by = ROOT
            }
        }
        
        area_bloodsail_hold = {
            add_core = A15
            change_culture = culture_kul_tiran
            change_religion = shadow
        }
        
        ai_chance = {
            factor = 10
            
            # VAWSER: if ROOT is AI and a subject, then it is likely a colonial nation and as such shouldn't cede away land.
            modifier = {
                factor = 0
                
                area_bloodsail_hold = {
                    owner = {
                        is_subject = yes
                    }
                }
            }
        }
	}
    
    after = {
        A15 = {
            country_event = { id = wwu_setup_pirate.2 days = 3 }
        }
    }
}
