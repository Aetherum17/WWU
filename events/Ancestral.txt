namespace = ancestral


country_event = {
	id = ancestral.2
	title = passing_of_great_leader.title
	desc = passing_of_great_leader.desc # $MONARCH$ is a great leader of our people and almost a legendary figure amongst the [Root.GetCulture] rulers. Sadly, recently [Root.Monarch.GetSheHeCap] got terribly sick and shamans think [Root.Monarch.GetSheHeCap] might've left only few days.\n\nIn ancestral tradition, we know spirits of great shamans do not leave this world after-life, and sometimes new leaders channel their spirits for guidance. Since [Root.Monarch.GetSheHeCap] has proven [Root.Monarch.GetHerHis] countless of times, we think $MONARCH$ deserves a special tomb only for [Root.Monarch.GetHerHim] for the eternal rest. On the other hand, certain statesmen in our council are against it. Some of them do not think $MONARCH$ deserves such treatment and doing so might offend other deceased [Root.Monarch.GetTitle]s.
	picture = ELEMENTAL_eventPicture
	
	trigger = {
		religion = religion_ancestral
		ruler_religion = religion_ancestral
		months_of_ruling = 180
		ruler_age = 55
		is_great_leader = yes
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = {
	
	}
	
	option = {
		name = passing_of_great_leader.opt1
		#log = "passing_of_great_leader.opt1" # Build a massive tomb in memory of $MONARCH$ 
		custom_tooltip = passing_of_great_leader.opt1.tt
		hidden_effect = {
			exile_ruler_as = honored_great_person
			set_country_flag = honored_great_person_flag
			random = {
				chance = 50
				
				country_event = { id = ancestral.3 days = 3650 random = 1825 }
			}
		}
		tooltip = {
			kill_ruler = yes
		}
		add_years_of_income = -1
		add_prestige = 15
		random_list = {
			50 = {
				add_karma = -10
			}
			50 = {
				add_stability = 1
			}
		}
	}
	option = {
		name = passing_of_great_leader.opt2
		#log = "passing_of_great_leader.opt2" # [Root.Monarch.GetSheHeCap] is going near the other [Root.Monarch.GetTitle]s before [Root.Monarch.GetHerHim]!
		custom_tooltip = passing_of_great_leader.opt2.tt
		hidden_effect = {
			exile_ruler_as = disrespected_great_person
			set_country_flag = disrespected_great_person_flag
			random = {
				chance = 50
				
				country_event = { id = ancestral.4 days = 3650 random = 1825 }
			}
		}
		tooltip = {
			kill_ruler = yes
		}
		add_karma = 10
	}
	option = {
		name = passing_of_great_leader.opt3
		#log = "passing_of_great_leader.opt3" # Build [Root.Monarch.GetHerHis]'s statue
		custom_tooltip = passing_of_great_leader.opt3.tt
		kill_ruler = yes
		add_years_of_income = -0.5
		add_prestige = 5
	}
}

	}

		change_dip = 1
		change_mil = 1
	}

		add_stability = 1
		add_years_of_income = 1
		add_manpower = 10
	}
	option = {
		name = help_of_old_spirit.opt4

	}

	}
