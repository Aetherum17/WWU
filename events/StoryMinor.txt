# Balnazzar's Reveal
country_event = {
	id = story.1
	title = story.1.title
	desc = story.1.desc
	picture = NATHREZIM_eventPicture
	#goto = root
	
	major = yes
	
	trigger = {
		always = no
		tag = A22 # Scarlet Crusade
		OR = {
			NOT = { stability = -2 }
			dominant_culture = culture_forsaken
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = story.1.option.a
		log = "story.1.option.a"
		
        ##if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - story.1.option.a" }

		change_tag = X01 # The Risen
		change_government = dictatorship
		change_primary_culture = culture_forsaken
		change_religion = religion_fel
		every_owned_province = {
			change_culture = culture_forsaken
			change_religion = religion_fel
		}
		add_stability = 6
		
		if = {
			limit = {	
				1011 = {
					NOT = { owned_by = ROOT }
				}
			}
			1011 = {
				cede_province = ROOT
			}
		}
		
		if = {
			limit = {	
				1028 = {
					NOT = { owned_by = ROOT }
				}
			}
			1028 = {
				cede_province = ROOT
			}
		}
		
		
		if = {
			limit = {	
				519 = {
					NOT = { owned_by = ROOT }
				}
			}
			519 = {
				cede_province = ROOT
			}
		}
		
		if = {
			limit = {	
				539 = {
					NOT = { owned_by = ROOT }
				}
			}
			539 = {
				cede_province = ROOT
			}
		}
		
		define_ruler = {
			name = "Balnazzar"
			dynasty = ""
			ADM = 3
			DIP = 1
			MIL = 6
			attach_leader = "Balnazzar"
		}
	}
}

# Insidious Happenings 
country_event = {
	id = story.2
	title = story.2.title
	desc = story.2.desc
	picture = FORSAKEN_eventPicture
	#goto = root
	
	trigger = {
		tag = A22 # Scarlet Crusade
		NOT = { stability = 0 }
		any_owned_province = {
			is_capital = no
			NOT = { culture = culture_forsaken }
		}
		always = no
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	option = {
		name = story.2.option.a
		log = "story.2.option.a"
		
        ##if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - story.2.option.a" }

		random_owned_province = {
			limit = { is_capital = no }
		
			change_culture = culture_forsaken
		}
	}
}
