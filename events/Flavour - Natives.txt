# Province Event Manager
country_event = {
	id = wwu_natives.100
	title = wwu_natives.100.title
	desc = wwu_natives.100.desc
	picture = JUNGLE_GIANTS_eventPicture
    
    is_triggered_only = yes
    
	trigger = {
        any_owned_province = {
            is_colony = yes
        }
	}
	
	mean_time_to_happen = {
		days = 1
	}
    
    immediate = {
        hidden_effect = {
            random_owned_province = {
                limit = { is_colony = yes }
                save_event_target_as = target_province
            }
        }
    }
    
	option = {
		name = wwu_natives.1.option.a
        
        event_target:target_province = {
            random_list = {
                10 = { province_event = { id = wwu_natives.1 } }
                10 = { province_event = { id = wwu_natives.2 } }
                10 = { province_event = { id = wwu_natives.3 } }
                10 = { province_event = { id = wwu_natives.4 } }

            }
        }
	}
}

# Natives Encountering our Colonies
province_event = {
	id = wwu_natives.1
	title = wwu_natives.1.title
	desc = wwu_natives.1.desc
	picture = JUNGLE_GIANTS_eventPicture
	goto = ROOT
    
    is_triggered_only = yes
    
	trigger = {
		NOT = { has_dlc = "The Cossacks" }
		native_size = 3
		owner = {
			NOT = { has_country_flag = NF_colony }
		}
		is_colony = yes
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = wwu_natives.1.option.a
		
		ai_chance = { factor = 40 }
		owner = {
			set_country_flag = NF_colony
			set_country_flag = NF_ruthless
		}
		add_colonysize = 200
		change_native_size = -3
		custom_tooltip = native_ruthless_tt
	}
	option = {
		name = wwu_natives.1.option.b
		
		ai_chance = { factor = 20 }
		owner = {
			add_treasury = -50
			set_country_flag = NF_colony
			set_country_flag = NF_trader
		}
		add_base_tax = 1
		custom_tooltip = native_trader_tt
	}	
	option = {
		name = wwu_natives.1.option.c
		
		ai_chance = { factor = 20 }
		owner = {
			add_prestige = -2
			set_country_flag = NF_colony
			set_country_flag = NF_peacenik		
		}
		custom_tooltip = native_peace_tt
	}	
}
