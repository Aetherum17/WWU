# Criminal Acumen (Spying) - Mechanic
country_event = {
	id = wwu_government_mechanics.3
	title = wwu_government_mechanics.3.title
	desc = wwu_government_mechanics.3.desc
	picture = ASSASSINS_BLADE_eventPicture
	
    hidden = yes
    
	trigger = {
        has_government_mechanic = criminal_acumen_mechanic
        
        any_country = {
            has_spy_network_from = {
                who = ROOT
                value = 30
            }
        }
	}
	
	mean_time_to_happen = {
		months = 12
        
        modifier = {
            factor = 0.8
            
            calc_true_if = {
                amount = 3
                all_country = {
                    has_spy_network_from = {
                        who = ROOT
                        value = 30
                    }
                }
            }
        }
        
        modifier = {
            factor = 0.8
            
            calc_true_if = {
                amount = 5
                all_country = {
                    has_spy_network_from = {
                        who = ROOT
                        value = 30
                    }
                }
            }
        }
	}
	
	option = {
		name = wwu_government_mechanics.3.option.a
		
		add_government_power = {
            mechanic_type = criminal_acumen_mechanic
            power_type = criminal_acumen
            value = 5
        }
	}
}

# War Support - Add flag for war win
country_event = {
	id = wwu_government_mechanics.10
	title = wwu_government_mechanics.10.title
	desc = wwu_government_mechanics.10.desc
	picture = ASSASSINS_BLADE_eventPicture
	
    hidden = yes
    
	trigger = {
        NOT = { has_country_flag = war_support_faction_leader }
        OR = {
            AND = {
                has_country_modifier = head_of_alliance
                has_government_mechanic = faction_authority_alliance_mechanic
                is_at_war = yes
              
                any_country = {
                    has_country_modifier = member_of_alliance
                    
                    OR = {
                        is_in_war = {
                            attackers = THIS
                            attackers = ROOT
                        }
                        is_in_war = {
                            defenders = THIS
                            defenders = ROOT
                        }
                    }
                }
