namespace = random_org
# Military Guilds
country_event = {
	id = random_org.1
	title = war_campaign.title
	desc = war_campaign.desc
	picture = GUILD_eventPicture
	
	mean_time_to_happen = { months = 540 }
	
	trigger = {
		any_country = {
			any_owned_province = { region = ROOT }
			has_country_modifier = is_evil
		}
	}
	
	immediate = {
		random_country = {
			limit = { has_country_modifier = is_evil }
			save_global_event_target_as = target_of_guilds
		}
	}
	
	option = {
		name = war_campaign.opt1
		add_dip_power = -100
		set_country_flag = can_have_coalition@event_target:target_of_guilds
		country_event = { id = random_org.3 days = 365 }
		custom_tooltip = war_campaign.opt1.tt
		every_country = {
			limit = { 
				is_organization = yes
				has_opinion = { who = ROOT value = -50 }
				NOT = { tag = ROOT }
			}
			country_event = { id = random_org.2 }
		}
	}
	
	option = {
		name = war_campaign.opt2
	}
	
}
