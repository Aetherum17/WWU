namespace = enslavement

country_event = {
	id = enslavement.1
	title = enslavement.1.title
	desc = enslavement.1.desc
	picture = ENSLAVEMENT_eventPicture
	
	hidden = no
	
    mean_time_to_happen = {
		months = 1
	}
	
	trigger = {
		any_country = {
			ROOT = {
				has_casus_belli = {
					type = cb_enslavement
					target = PREV
				}
				war_with = PREV
				NOT = { has_country_flag = has_enslavement_war@PREV }
			}
		}
	}
	
    option = {
        name = enslavement.1.opt1
		random_country = {
			limit = {
				ROOT = {
					has_casus_belli = {
						type = cb_enslavement
						target = PREV
					}
					war_with = PREV
				}
			}
			save_event_target_as = target_of_enslavement
		}
		event_target:target_of_enslavement = { }
		set_country_flag = has_enslavement_war@event_target:target_of_enslavement
    }
}

country_event = {
	id = enslavement.2
	title = enslavement.2.title
	desc = enslavement.2.desc
	picture = ENSLAVEMENT_eventPicture
	
    is_triggered_only = yes
	hidden = no
	
	trigger = {
		any_subject_country = {
			ROOT = {
				has_country_flag = has_enslavement_war@PREV
			}
		}
	}
	
    option = {
        name = enslavement.2.opt1
		random_subject_country = {
			limit = {
				ROOT = {
					has_country_flag = has_enslavement_war@PREV
				}
			}
			ROOT = {
				create_subject = {
					subject_type = enslavement
					subject = PREV
				}
			}
		}
	}
}
