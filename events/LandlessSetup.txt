namespace = landless
# Startup
country_event = {
	id = landless.1
	title = landless.1.title
	desc = landless.1.desc
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		is_landless = yes
	}
	option = {
		name = landless.1.opt1
		random_province = {
			limit = { reserved_for_landless = yes }
			cede_province = ROOT
			add_core = ROOT
		}
		every_country = {
			add_opinion = {
				who = ROOT
				modifier = landless_opinion
			}
			reverse_add_opinion = {
				who = ROOT
				modifier = landless_opinion
			}
		}
		define_ruler_to_general = {
			fire = 3
			shock = 3
			manuever = 3
			siege = 3
		}
	}
}
# Defeat
country_event = {

	title = landless.1.title
	desc = landless.1.desc
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		is_landless = yes
		NOT = { army_size_percentage = 0.01 }
	}
	option = {
		name = landless.1.opt1
		every_owned_province = {
			cede_province = '---'
		}
	}
}
