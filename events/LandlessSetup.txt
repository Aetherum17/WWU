namespace = landless
# Startup
country_event = {
	id = landless.1
	title = landless.1.title
	desc = landless.1.desc
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		is_landless = yes
	}
	option = {
		name = landless.1.opt1
		random_province = {
			limit = { reserved_for_landless = yes }
			cede_province = ROOT
			add_core = ROOT
		}
		capital_scope = {
			cede_province = '---'
		}
		every_country = {
			add_opinion = {
				who = ROOT
				modifier = landless_opinion
			}
			reverse_add_opinion = {
				who = ROOT
				modifier = landless_opinion
			}
		}
		define_ruler_to_general = {
			fire = 3
			shock = 3
			manuever = 3
			siege = 3
		}
	}
}
# Discovery
country_event = {
	id = landless.2
	title = landless.2.title
	desc = landless.2.desc
	hidden = yes
	
	trigger = {
		is_landless = yes
		NOT = { any_province = {
			num_of_units_in_province = {
					who = ROOT 
					amount = 1
			}
			all_neighbor_province = {
				has_discovered = ROOT
			}
		} }
	}
	
	mean_time_to_happen = {
		days = 15
	}
	
	option = {
		name = landless.1.opt1
		random_province = {
			limit = {
				num_of_units_in_province = {
					who = ROOT 
					amount = 1
				}
			}
			every_neighbor_province = {
				discover_country = ROOT 
			}
		}
	}
}
