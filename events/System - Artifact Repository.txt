# Artifact Heist
country_event = {
	id = wwu_artifact_repository.2
	title = wwu_artifact_repository.2.title
	desc = wwu_artifact_repository.2.desc
	picture = BIG_BOOK_eventPicture
	
	trigger = {
        has_country_flag = artifact_heist_active
        any_country = {
            has_country_flag = artifact_heist_target
        }
    }
    
    mean_time_to_happen = {
		days = 1
	}
    
    immediate = {
        hidden_effect = {
            clr_country_flag = artifact_heist_active
            
            every_country = {
                limit = {
                    has_country_flag = artifact_heist_target
                }
                save_event_target_as = artifact_heist_target
            }
        }
    }
    
    option = {
		name = wwu_artifact_repository.2.option.apocalypse
        
        trigger = {
            event_target:artifact_heist_target = {
                has_country_flag = artifact_apocalypse
            }
        }
        
        steal_artifact = {
            artifact = apocalypse
        }
	}
    option = {
		name = wwu_artifact_repository.2.option.twinblades_of_the_deceiver
        
        trigger = {
            event_target:artifact_heist_target = {
                has_country_flag = artifact_twinblades_of_the_deceiver
            }
        }

        }
	}
    option = {
