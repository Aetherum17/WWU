namespace = wwu_setup_demonic_vanguard

# Demonic Vanguard
# This nation represents Archimonde and his forces attempting to destroy Nordrassil
country_event = {
	id = wwu_setup_demonic_vanguard.1
	title = wwu_setup_demonic_vanguard.1.title
	desc = wwu_setup_demonic_vanguard.1.desc
	picture = HYJAL_ASSAULT_eventPicture

    major = yes
	fire_only_once = yes
    
	trigger = {
        OR = {
            AND = {
                ai = no
                has_country_flag = host_player
            }
            is_great_power = yes
        }
        NOT = { exists = DVG }
        NOT = { has_global_flag = setup_demonic_vanguard }
        is_year = 615
        NOT = { is_year = 618 }
    }
    
    immediate = {
        hidden_effect = {
            set_global_flag = setup_demonic_vanguard
        }
    }
    
    mean_time_to_happen = {
		months = 12
	}

	option = {
		name = wwu_setup_demonic_vanguard.1.option.a
        
        area_upper_groves = {
            cede_province = DVG
            add_core = DVG
            change_culture = culture_demonspawn
            change_religion = fel
        }
        
        area_nordrassil = {
            cede_province = DVG
            add_core = DVG
            change_culture = culture_demonspawn
            change_religion = fel
        }
        
        1334 = {
            change_culture = culture_eredar
        }
        
        DVG = {
            set_capital = 1334
        }
	}
    
    after = {
        DVG = {
            define_ruler = {
                name = "Archimonde" 
                dynasty = "the Defiler" 
                adm = 7
                dip = 7
                mil = 7
                age = 500
            }
        }
    }
}
