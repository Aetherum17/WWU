    is_triggered_only = yes
    
	trigger = {
		owner = { is_free_or_tributary_trigger = yes }
		any_neighbor_province = {
			NOT = { owned_by = ROOT }
			NOT = { is_core = ROOT }
			NOT = { is_claim = ROOT }
			has_discovered = ROOT
			is_capital = no
			owner = {
				NOT = { is_subject_of = ROOT }
			}
		}
	}

	mean_time_to_happen = {
		days = 1	
	}

	immediate = {
		hidden_effect = {
			random_neighbor_province = {
				limit = {
					NOT = { owned_by = ROOT }
					NOT = { is_core = ROOT }
					NOT = { is_claim = ROOT }
					has_discovered = ROOT
					is_capital = no
					NOT = { owner = { is_subject_of = ROOT } }
				}
				save_event_target_as = claim_province
			}
		}
	}

	is_mtth_scaled_to_size = 1

	option = {
		name = "EVTOPTA5087" # Press issue
		event_target:claim_province = {
			add_claim = ROOT
			owner = {
				add_casus_belli = {	
					target = ROOT
					type = cb_insult
					months = 36
				}
				add_opinion = { who = ROOT modifier = opinion_press_issue }
			}
		}
	}
	option = {
		name = "EVTOPTB5087" # Settle issue
		owner = { add_prestige = -5 }
	}
