	desc = wwu_zandalar_naga_invasion.10.desc
	picture = HUNGRY_NAGA_eventPicture

            # Covescale Rise
            2616 = {
                OR =  {
                    colonysize = 20
                    owned_by = ROOT
                }
            }
        }
	}
    
    mean_time_to_happen = { 
        days = 1
    }
    
    immediate = {
        hidden_effect = {
            set_global_flag = nazeshi_invasion

            # Backwater Beach
            if = {
                limit = {
                    2745 = {
                        OR =  {
                            is_empty = yes
                            colonysize = 20
                            owned_by = ROOT
                        }
                    }
                }
                2745 = {
                    cede_province = ZNI
                    add_core = ZNI
                }
            }
            # Seeker's Outpost
            if = {
                limit = {
                    2744 = {
                        OR =  {
                            is_empty = yes
                            colonysize = 20
                            owned_by = ROOT
                        }
                    }
                }
                2744 = {
                    cede_province = ZNI
                    add_core = ZNI
                }
            }
            # Dreadpearl Shallows
            if = {
                limit = {
                    1133 = {
                        OR =  {
                            is_empty = yes
                            colonysize = 20
                            owned_by = ROOT
                        }
                    }
                }
                1133 = {
                    cede_province = ZNI
                    add_core = ZNI
                }
            }
            # Covescale Rise
            if = {
                limit = {
                    2616 = {
                        OR =  {
                            is_empty = yes
                            colonysize = 20
                            owned_by = ROOT
                        }
                    }
                }
                2616 = {
                    cede_province = ZNI
                    add_core = ZNI
                }
            }
            
            ZNI = {
                capital_scope = {
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI

                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                    infantry = ZNI
                }
                
                add_country_modifier = {
                    name = zandalar_naga_invasion
                    duration = -1
                }
                
                create_general = { tradition = 70 }
                
                add_treasury = 2000
                add_manpower = 30
            }
            
            zandalar_superregion = {
                discover_country = ZNI
            }
        }
    }
    
    option = {
        name = wwu_zandalar_naga_invasion.10.option.a
        
        custom_tooltip = zandalari_nazeshi_invasion_tt
        
        ZNI = {
            declare_war_with_cb = {
                who = ROOT
                casus_belli = cb_naga_invasion_on_zandalar
            }
        }
    }

    option = {
        name = wwu_zandalar_naga_invasion.10.option.b
        
        custom_tooltip = zandalari_nazeshi_ignore_tt
        
        add_prestige = -10
        add_legitimacy = -10
        
        if = {
            limit = {   
                tag = A78 # Zandalar
                has_country_flag = has_council_of_zanchul
            }
            increase_zul_power_by_5 = yes
        }
    }
}

    fire_only_once = yes

    }

            }
            decrease_zul_power_by_5 = yes
        }
    }
}
