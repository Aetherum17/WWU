# Withering
country_event = {
	id = wwu_loss_of_the_nightwell.10
	title = wwu_loss_of_the_nightwell.10.title
	desc = wwu_loss_of_the_nightwell.10.desc
	picture = SURAMAR_eventPicture

    is_triggered_only = yes
    
    trigger = {
        NOT = { has_country_flag = postponed_arcwine_draught }
    }

	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		random_owned_province = {
			limit = {
				is_capital = no
				culture = culture_nightborne
			}
			
			save_event_target_as = withering_1
			set_province_flag = withering_flag
		}
        
		random_owned_province = {
			limit = {
				is_capital = no
				culture = culture_nightborne
				NOT = { has_province_flag = withering_flag }
			}
			
			save_event_target_as = withering_2
			set_province_flag = withering_flag
		}
		
	}

	option = {
		name = wwu_loss_of_the_nightwell.10.option.a
        
		trigger = {
			has_saved_event_target = withering_1
		}
		
		add_adm_power = -25
        
		event_target:withering_1 = {
			add_base_production = 1
		}
        
		event_target:withering_2 = {
			add_base_production = -1
			change_culture = culture_withered
			add_province_modifier = {
				name = withered_province
				duration = 730
			}
		}
        
	}
    
    option = {
		name = wwu_loss_of_the_nightwell.10.option.b
        
		trigger = {
			has_saved_event_target = withering_2 # Steer arcwine supplies to [withering_2.GetName] and let [withering_1.GetName] wither.
		}
		
		add_adm_power = -25
        
		event_target:withering_2 = {
			add_base_production = 1
		}
        
		event_target:withering_1 = {
			add_base_production = -1
			change_culture = culture_withered
			add_province_modifier = {
				name = withered_province
				duration = 730
			}
		}
	}
    
    option = {
		name = wwu_loss_of_the_nightwell.10.option.c
        
		trigger = {
			has_ruler_flag = mage_personality
		}
		
		add_mil_power = -50
		add_dip_power = -50
		
        custom_tooltip = REDUCE_WITHERING_RATE_TT
        hidden_effect = {
            set_country_flag  = postponed_arcwine_draught
        }
	}
	
	after = {
		event_target:withering_1 = {
			clr_province_flag = withering_flag
		}
		event_target:withering_2 = {
			clr_province_flag = withering_flag
		}
	}
}
