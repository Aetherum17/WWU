# Umbrafen
country_event = {
	id = wwu_setup_draenei.1
	title = wwu_setup_draenei.1.title
	desc = wwu_setup_draenei.1.desc
	picture = ZANGERMARSH_eventPicture

	is_triggered_only = yes
    
	trigger = {
        NOT = { exists = F65 }
        NOT = { has_global_flag = setup_umbrafen }
        NOT = { has_global_flag = outland_country_setup }
        
        had_global_flag = {
            flag = setup_outland
            days = 365
        }
        
        if = {
            limit = {
                ai = no
                capital_scope = {
                    region = region_zangarmarsh
                }
            }
            
            ai = no
        }
        
        OR = {
            owns = 1608
            owns = 1606
            owns = 1607
            owns = 1610
        }
    }
    
    immediate = {
        hidden_effect = {
            set_global_flag = setup_umbrafen
        }
    }
    
    mean_time_to_happen = {
		days = 1
	}

	option = {
		name = wwu_setup_draenei.1.option.a
        
        spawn_nation_by_province = { province = 1608 tag = F65 culture = culture_lost_one religion = shadow }
        spawn_nation_by_province = { province = 1606 tag = F65 culture = culture_lost_one religion = shadow }
        spawn_nation_by_province = { province = 1607 tag = F65 culture = culture_lost_one religion = shadow }
        spawn_nation_by_province = { province = 1610 tag = F65 culture = culture_lost_one religion = shadow }
        
        ai_chance = {
            factor = 100
        }
	}
	option = {
		name = wwu_setup_draenei.1.option.b
        
        add_dip_power = -100
        
        custom_tooltip = PARTIAL_SETUP_TT
        
        spawn_nation_by_province_conditional = { province = 1608 tag = F65 culture = culture_lost_one religion = shadow }
